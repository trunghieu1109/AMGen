    cot_instruction_2_1 = "Sub-task 2_1: Combine the geometric relationships and numeric side lengths to set up explicit equations involving the length CE and other segment lengths on the line containing D, E, C, F. Prepare these equations for algebraic solving, ensuring that all possible orderings and roots from previous subtasks are included in the formulation."
    cot_agent_2_1 = LLMAgentBase(["thinking", "answer"], "Chain-of-Thought Agent", model=self.node_model, temperature=0.0)
    subtask_desc_2_1 = {
        "subtask_id": "subtask_2_1",
        "instruction": cot_instruction_2_1,
        "context": ["user query", thinking_1_2.content, thinking_1_3.content],
        "agent_collaboration": "CoT"
    }
    thinking_2_1, answer_2_1 = await cot_agent_2_1([taskInfo, thinking_1_2, thinking_1_3], cot_instruction_2_1, is_sub_task=True)
    agents.append(f"CoT agent {cot_agent_2_1.id}, set up equations for CE, thinking: {thinking_2_1.content}; answer: {answer_2_1.content}")
    sub_tasks.append(f"Sub-task 2_1 output: thinking - {thinking_2_1.content}; answer - {answer_2_1.content}")
    subtask_desc_2_1['response'] = {"thinking": thinking_2_1, "answer": answer_2_1}
    logs.append(subtask_desc_2_1)
    print("Step 2_1: ", sub_tasks[-1])

    cot_sc_instruction_2_2a = "Sub-task 2_2a: Solve the key equation(s) algebraically to find all possible values for the coordinate(s) of point E and the length CE. Explicitly retain both roots (positive and negative) or all algebraic solutions without discarding any based on intuition or incomplete reasoning."
    cot_sc_agents_2_2a = [LLMAgentBase(["thinking", "answer"], "Chain-of-Thought Agent", model=self.node_model, temperature=0.0) for _ in range(self.max_sc)]
    possible_answers_2_2a = []
    possible_thinkings_2_2a = []
    subtask_desc_2_2a = {
        "subtask_id": "subtask_2_2a",
        "instruction": cot_sc_instruction_2_2a,
        "context": ["user query", thinking_2_1.content],
        "agent_collaboration": "SC_CoT"
    }
    for i in range(self.max_sc):
        thinking_2_2a, answer_2_2a = await cot_sc_agents_2_2a[i]([taskInfo, thinking_2_1], cot_sc_instruction_2_2a, is_sub_task=True)
        agents.append(f"CoT-SC agent {cot_sc_agents_2_2a[i].id}, solve algebraic equations for CE, thinking: {thinking_2_2a.content}; answer: {answer_2_2a.content}")
        possible_answers_2_2a.append(answer_2_2a)
        possible_thinkings_2_2a.append(thinking_2_2a)
    final_decision_agent_2_2a = LLMAgentBase(["thinking", "answer"], "Final Decision Agent", model=self.node_model, temperature=0.0)
    thinking_2_2a, answer_2_2a = await final_decision_agent_2_2a([taskInfo] + possible_thinkings_2_2a, "Sub-task 2_2a: Synthesize and choose the most consistent algebraic solutions for CE.", is_sub_task=True)
    sub_tasks.append(f"Sub-task 2_2a output: thinking - {thinking_2_2a.content}; answer - {answer_2_2a.content}")
    subtask_desc_2_2a['response'] = {"thinking": thinking_2_2a, "answer": answer_2_2a}
    logs.append(subtask_desc_2_2a)
    print("Step 2_2a: ", sub_tasks[-1])

    debate_instruction_2_2b = "Sub-task 2_2b: Perform a geometric validation of each algebraic solution obtained in subtask 2_2a. Critically assess the feasibility of each root with respect to the orderings of points D, E, C, F, the rectangle side lengths, and the concyclicity condition. Use a Debate pattern where agents argue for or against each candidate solution based on geometric consistency and problem constraints. Given solutions to the problem from other agents, consider their opinions as additional advice. Please think carefully and provide an updated answer."
    debate_agents_2_2b = [LLMAgentBase(["thinking", "answer"], "Debate Agent", model=self.node_model, role=role, temperature=0.0) for role in self.debate_role]
    N_max_2_2b = self.max_round
    all_thinking_2_2b = [[] for _ in range(N_max_2_2b)]
    all_answer_2_2b = [[] for _ in range(N_max_2_2b)]
    subtask_desc_2_2b = {
        "subtask_id": "subtask_2_2b",
        "instruction": debate_instruction_2_2b,
        "context": ["user query", thinking_2_2a.content, thinking_1_2.content, thinking_1_3.content],
        "agent_collaboration": "Debate"
    }
    for r in range(N_max_2_2b):
        for i, agent in enumerate(debate_agents_2_2b):
            if r == 0:
                thinking_2_2b, answer_2_2b = await agent([taskInfo, thinking_2_2a, thinking_1_2, thinking_1_3], debate_instruction_2_2b, r, is_sub_task=True)
            else:
                input_infos_2_2b = [taskInfo, thinking_2_2a, thinking_1_2, thinking_1_3] + all_thinking_2_2b[r-1]
                thinking_2_2b, answer_2_2b = await agent(input_infos_2_2b, debate_instruction_2_2b, r, is_sub_task=True)
            agents.append(f"Debate agent {agent.id}, round {r}, validate algebraic roots geometrically, thinking: {thinking_2_2b.content}; answer: {answer_2_2b.content}")
            all_thinking_2_2b[r].append(thinking_2_2b)
            all_answer_2_2b[r].append(answer_2_2b)
    final_decision_agent_2_2b = LLMAgentBase(["thinking", "answer"], "Final Decision Agent", model=self.node_model, temperature=0.0)
    thinking_2_2b, answer_2_2b = await final_decision_agent_2_2b([taskInfo] + all_thinking_2_2b[-1], "Sub-task 2_2b: Given all the above thinking and answers, reason over them carefully and provide a final answer.", is_sub_task=True)
    sub_tasks.append(f"Sub-task 2_2b output: thinking - {thinking_2_2b.content}; answer - {answer_2_2b.content}")
    subtask_desc_2_2b['response'] = {"thinking": thinking_2_2b, "answer": answer_2_2b}
    logs.append(subtask_desc_2_2b)
    print("Step 2_2b: ", sub_tasks[-1])

    reflect_inst_2_3 = "Given previous attempts and feedback, carefully consider where you could go wrong in your latest attempt. Using insights from previous attempts, try to solve the task better."
    cot_reflect_instruction_2_3 = "Sub-task 2_3: Cross-verify the accepted solution(s) for length CE against all given geometric constraints, including collinearity, concyclicity, rectangle properties, and side lengths. Explicitly document the reasoning for acceptance or rejection of each candidate and finalize the length CE only after thorough validation." + reflect_inst_2_3
    cot_agent_2_3 = LLMAgentBase(["thinking", "answer"], "Chain-of-Thought Agent", model=self.node_model, temperature=0.0)
    critic_agent_2_3 = LLMAgentBase(["feedback", "correct"], "Critic Agent", model=self.node_model, temperature=0.0)
    N_max_2_3 = self.max_round
    cot_inputs_2_3 = [taskInfo, thinking_2_2b, thinking_1_2, thinking_1_3]
    subtask_desc_2_3 = {
        "subtask_id": "subtask_2_3",
        "instruction": cot_reflect_instruction_2_3,
        "context": ["user query", thinking_2_2b.content, thinking_1_2.content, thinking_1_3.content],
        "agent_collaboration": "Reflexion"
    }
    thinking_2_3, answer_2_3 = await cot_agent_2_3(cot_inputs_2_3, cot_reflect_instruction_2_3, 0, is_sub_task=True)
    agents.append(f"Reflexion CoT agent {cot_agent_2_3.id}, cross-verify solutions for CE, thinking: {thinking_2_3.content}; answer: {answer_2_3.content}")
    critic_inst_2_3 = "Please review the answer above and criticize on where might be wrong. If you are absolutely sure it is correct, output exactly 'True' in 'correct'"
    for i in range(N_max_2_3):
        feedback_2_3, correct_2_3 = await critic_agent_2_3([taskInfo, thinking_2_3], "Please review and provide the limitations of provided solutions" + critic_inst_2_3, i, is_sub_task=True)
        agents.append(f"Critic agent {critic_agent_2_3.id}, providing feedback, thinking: {feedback_2_3.content}; answer: {correct_2_3.content}")
        if correct_2_3.content == "True":
            break
        cot_inputs_2_3.extend([thinking_2_3, feedback_2_3])
        thinking_2_3, answer_2_3 = await cot_agent_2_3(cot_inputs_2_3, cot_reflect_instruction_2_3, i + 1, is_sub_task=True)
        agents.append(f"Reflexion CoT agent {cot_agent_2_3.id}, refining valid scenarios for CE, thinking: {thinking_2_3.content}; answer: {answer_2_3.content}")
    sub_tasks.append(f"Sub-task 2_3 output: thinking - {thinking_2_3.content}; answer - {answer_2_3.content}")
    subtask_desc_2_3['response'] = {"thinking": thinking_2_3, "answer": answer_2_3}
    logs.append(subtask_desc_2_3)
    print("Step 2_3: ", sub_tasks[-1])

    final_answer = await self.make_final_answer(thinking_2_3, answer_2_3, sub_tasks, agents)
    return final_answer, logs
